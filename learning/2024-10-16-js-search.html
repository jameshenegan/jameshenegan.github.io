Vanilla JavaScript search

Here’s a step-by-step tutorial to help you implement a searchable and paginated table with persistent search and filtering using vanilla JavaScript, HTML, and CSS. This tutorial will walk you through:

  • Creating a configuration-based dynamic table
  • Implementing search functionality
  • Adding filtering through dropdowns
  • Ensuring the search and filters persist across pages

Step 1: Set Up the Basic HTML Structure

We’ll start by creating a simple HTML structure that includes:

  1. A section for dropdown filters (which we will generate dynamically).
  2. A search input for text-based filtering.
  3. A table that will display the filtered and paginated results.
  4. Pagination controls (buttons for navigating between pages).

Step 2: Create a Configuration Object

In our table, we have multiple columns like Table Category, Table Name, and so on. We want our app to be flexible so that if the column names change or if new columns are added, the table still works without us needing to update the code manually.

We’ll define a configuration object in our JavaScript file (app.js) that stores information about how the table should be generated, including which columns are categorical (used for dropdown filters).

Here’s an example of the configuration object:

const config = {
  tableAttributes: [
    { key: "Table Category", displayName: "Category", isCategorical: true },
    { key: "Table Name", displayName: "Table", isCategorical: true },
    { key: "Variable Name", displayName: "Column", isCategorical: false },
    {
      key: "Description",
      displayName: "Column Description",
      isCategorical: false,
    },
  ],
  rowsPerPage: 5,
};

In this configuration:

  • key: Refers to the actual key in the JSON data.
  • displayName: Refers to the label we want to show in the table headers.
  • isCategorical: Specifies whether this column should have a dropdown filter.

Step 3: Load and Render the Table

Now we’ll write JavaScript to dynamically generate the table headers and rows based on the configuration. We’ll also include logic to render only the number of rows specified in the configuration (rowsPerPage) and handle pagination.

Here’s the basic logic for generating the table:

// Sample metadata for demonstration purposes
const metadata = [
  {
    "Table Category": "User Data",
    "Table Name": "Users",
    "Variable Name": "UserID",
    Description: "Unique ID for each user",
  },
  {
    "Table Category": "Order Data",
    "Table Name": "Orders",
    "Variable Name": "OrderID",
    Description: "Unique ID for each order",
  },
  // More rows can be added here...
];

let currentPage = 1;
let filteredMetadata = [...metadata]; // Stores the currently filtered results

// Create the table headers based on the configuration
function createTableHeaders() {
  const headerRow = document.getElementById("tableHeaders");
  headerRow.innerHTML = "";

  config.tableAttributes.forEach((attr) => {
    const th = document.createElement("th");
    th.textContent = attr.displayName; // Use the display name from the config
    headerRow.appendChild(th);
  });
}

// Render the metadata table and handle pagination
function renderTable(page = 1, data = filteredMetadata) {
  const tbody = document.getElementById("metadataBody");
  tbody.innerHTML = ""; // Clear previous rows

  const start = (page - 1) * config.rowsPerPage;
  const end = start + config.rowsPerPage;
  const pageData = data.slice(start, end);

  // Display the result count
  document.getElementById(
    "resultCount"
  ).textContent = `Showing ${data.length} results`;

  // Insert rows into the table
  pageData.forEach((item) => {
    const row = document.createElement("tr");

    config.tableAttributes.forEach((attr) => {
      const td = document.createElement("td");
      td.textContent = item[attr.key] || ""; // Use the key from the config to access the value
      row.appendChild(td);
    });

    tbody.appendChild(row);
  });

  // Update pagination info
  document.getElementById("pageInfo").textContent = `Page ${page}`;
  document.getElementById("prevBtn").disabled = page === 1;
  document.getElementById("nextBtn").disabled = end >= data.length;
}

Step 4: Add Filtering Logic

We’ll add a search input and dropdowns to filter the data. The dropdowns will be generated dynamically for any columns marked as categorical in the configuration.

Dynamically Create Dropdowns

// Create dropdowns for categorical columns
function createDropdowns() {
  const dropdownsDiv = document.getElementById("dropdowns");
  dropdownsDiv.innerHTML = "";

  config.tableAttributes.forEach((attr) => {
    if (attr.isCategorical) {
      const uniqueValues = [...new Set(metadata.map((item) => item[attr.key]))];

      const label = document.createElement("label");
      label.textContent = `Filter by ${attr.displayName}: `;

      const select = document.createElement("select");
      select.id = attr.key;
      select.onchange = filterByCategory;

      const allOption = document.createElement("option");
      allOption.value = "all";
      allOption.textContent = `All ${attr.displayName}s`;
      select.appendChild(allOption);

      uniqueValues.forEach((value) => {
        const option = document.createElement("option");
        option.value = value;
        option.textContent = value;
        select.appendChild(option);
      });

      dropdownsDiv.appendChild(label);
      dropdownsDiv.appendChild(select);
    }
  });
}

Apply Search and Filters Together

We’ll also ensure that both the search and dropdown filters are applied together when filtering the data:

let currentSearch = ""; // Store the current search value

// Filter the table based on search and dropdown filters
function filterTable() {
  const input = document.getElementById("searchInput").value.toLowerCase();
  currentSearch = input;

  applyFilters(); // Apply both search and dropdown filters
}

// Filter by category (dropdowns)
function filterByCategory() {
  currentPage = 1; // Reset to page 1 when changing categories
  applyFilters(); // Apply filters based on both category and search
}

// Apply both search and category filters
function applyFilters() {
  let filtered = [...metadata];

  // Apply category filters
  config.tableAttributes.forEach((attr) => {
    if (attr.isCategorical) {
      const selectedValue = document.getElementById(attr.key).value;
      if (selectedValue !== "all") {
        filtered = filtered.filter((item) => item[attr.key] === selectedValue);
      }
    }
  });

  // Apply search filter
  filtered = filtered.filter((item) =>
    config.tableAttributes.some(
      (attr) =>
        item[attr.key] && item[attr.key].toLowerCase().includes(currentSearch)
    )
  );

  filteredMetadata = filtered;
  renderTable(currentPage, filteredMetadata);
}

Step 5: Add Pagination Controls

We’ll implement the pagination

controls (Next/Previous) to allow users to navigate through the filtered data:

function prevPage() {
  if (currentPage > 1) {
    currentPage--;
    applyFilters(); // Reapply filters when changing pages
  }
}

function nextPage() {
  if (currentPage * config.rowsPerPage < filteredMetadata.length) {
    currentPage++;
    applyFilters(); // Reapply filters when changing pages
  }
}

Step 6: Initialize the Table

Finally, we’ll initialize everything by creating the table headers, dropdowns, and rendering the initial table:

// Initial setup
createTableHeaders(); // Create table headers dynamically
createDropdowns(); // Create dropdowns for categorical columns
renderTable(); // Render the table with the full dataset
